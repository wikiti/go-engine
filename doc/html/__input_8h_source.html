12  enum keystate_t {key_keyup = -2, key_unpressed = -1, key_keydown = 1, key_pressed = 2};
13  /*
14  * key_keyup (-2) => Instante en el que se deja de presionar la tecla.
15  * key_unpressed (-1) => Tecla sin presionar.
16  * key_keydown ( 1) => Instante en el que se presiona la tecla.
17  * key_pressed ( 2) => Tecla presionada.
18  *
19  * Ciclo normal de estados: unpressed -> keydown -> pressed -> keyup -> unpressed -> ...
20  *
21  * Si el valor es menor que 0, se considera la tecla como "no activada".
22  * Si el valor es mayor que 0, se considera la tecla como "activada".
23  */
