454 {
455  map<string, CShader*>::iterator it = shaders.find(name);
456  if(it == shaders.end())
457  return NULL;
458 
459  if(it->second->link_status)
460  return it->second;

464  glLinkProgram(it->second->Program);
465 
466  // check link status
467  GLint status = GL_FALSE;
468  glGetProgramiv(it->second->Program, GL_LINK_STATUS, &status);
469  if (status != GL_TRUE)
470  {
471  // The link has failed, check log info

477  gSystem_Debug.error("Failed to link the shader \"%s\": %s", name.c_str(), infoLog);
478  free(infoLog);

480  return NULL;
481  }
482 
483  // check if the shader will run in the current OpenGL state
484  glValidateProgram(it->second->Program);

489  return NULL;
490  }
491 
492  it->second->link_status = true;
493  // the program has been loaded/linked successfully
494 
495  return it->second;

500 {
501  GLuint vertShader = 0, fragShader = 0, geomShader = 0, programShader = 0;
502  bool loadStatus;
503 
504  // load and compile vertex. geometric and fragment sources
505  loadStatus = LoadShader(name, GL_VERTEX_SHADER, vertexFile, vertShader);
506  loadStatus &= LoadShader(name, GL_FRAGMENT_SHADER, fragmentFile, fragShader);
507  // if geometry file is provided, load it
508  if (geometryFile != "")
509  loadStatus &= LoadShader(name, GL_GEOMETRY_SHADER_EXT, geometryFile, geomShader);
510 
511  if(!loadStatus)
512  return NULL;
513  else
514  {
515  programShader = glCreateProgram();
516  if (programShader == 0)
517  {
518  gSystem_Debug.error("From Shader Manager: Could not generate glProgram (Shader) for %s", name.c_str());
519  return NULL;
520  }
521 
522  // attach the vertex and fragment shader codes, and the geometric if available
523  glAttachShader(programShader, vertShader);

525  glAttachShader(programShader, fragShader);
526 
527  // the program has been loaded/linked successfully
528  CShader* output = new CShader;
529  output->SetVertShader(vertShader);
530  output->SetFragShader(fragShader);
531  output->SetGeomShader(geomShader);
